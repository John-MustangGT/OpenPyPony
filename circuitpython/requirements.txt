# CircuitPython Library Requirements for OpenPonyLogger
#
# Install all with: circup install -r requirements.txt
# Or individually: circup install <library>

# Core Hardware Drivers
adafruit_lis3dh              # LIS3DH accelerometer
adafruit_gps                 # GPS module (UART)
adafruit_displayio_ssd1306   # SSD1306 OLED display
adafruit_display_text        # Text rendering
adafruit_bitmap_font         # Font support
neopixel                     # NeoPixel LED control
adafruit_pcf8523             # PCF8523 Real-Time Clock
adafruit_register	     # Dependancy for pcf8523

# Optional: Configuration
#cptoml                       # TOML parser (optional, has simple fallback)
